# Scroll down to "Obtaining and Importing Data"
# Import my files into QIIME 2 and create the artifact
qiime tools import \
  --type EMPSingleEndSequences \
  --input-path emp-single-end-sequences \
  --output-path emp-single-end-sequences.qza
# Scroll down to Demultiplexing sequences (only do this step if you have one sequence file)
# Demultiplex sequencing reads
qiime demux emp-single \
  --i-seqs emp-single-end-sequences.qza \
  --m-barcodes-file sample-metadata.tsv \
  --m-barcodes-column barcode-sequence \
  --o-per-sample-sequences demux.qza \
  --o-error-correction-details demux-details.qza
# Creating a file for visualization (takes your demux file and outputs the visualization - will show us the quality of the sequence reads)
qiime demux summarize \
  --i-data demux.qza \
  --o-visualization demux.qzv
# To visualize the file
qiime tools view demux.qzv (would not let me visualize)

# Drag and drop file - demux.qzv
# Single end reads - one graph / Paired end reads - two graphs
# If using the browser, close the window, or if using the command line, hit 'q' to get your command prompt back
# Scroll down to 'Option 1: DADA2'
# Quality and control of reads based on histogram - Processing and trimming the sequencing reads (For this one, trimming the 5' end at 0 and the 3' end at 120 - these numbers will change based on your sequence and which section you want to keep based on quality)
qiime dada2 denoise-single \
  --i-demultiplexed-seqs demux.qza \
  --p-trim-left 0 \
  --p-trunc-len 120 \
  --o-representative-sequences rep-seqs-dada2.qza \
  --o-table table-dada2.qza \
  --o-denoising-stats stats-dada2.qza
