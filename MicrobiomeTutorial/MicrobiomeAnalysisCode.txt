# Import my files into QIIME 2 and create the artifact
qiime tools import \
  --type EMPSingleEndSequences \
  --input-path emp-single-end-sequences \
  --output-path emp-single-end-sequences.qza
# Scroll down to Demultiplexing sequences (only do this step if you have one sequence file)
# Demultiplex sequencing reads
qiime demux emp-single \
  --i-seqs emp-single-end-sequences.qza \
  --m-barcodes-file sample-metadata.tsv \
  --m-barcodes-column barcode-sequence \
  --o-per-sample-sequences demux.qza \
  --o-error-correction-details demux-details.qza
# Creating a file for visualization (takes your demux file and outputs the visualization - will show us the quality of the sequence reads)
qiime demux summarize \
  --i-data demux.qza \
  --o-visualization demux.qzv
# To visualize the file
qiime tools view demux.qzv (would not let me visualize)

# Drag and drop file - demux.qzv
# Single end reads - one graph / Paired end reads - two graphs
# If using the browser, close the window, or if using the command line, hit 'q' to get your command prompt back
# Scroll down to 'Option 1: DADA2'
# Quality and control of reads based on histogram - Processing and trimming the sequencing reads (For this one, trimming the 5' end at 0 and the 3' end at 120 - these numbers will change based on your sequence and which section you want to keep based on quality)
qiime dada2 denoise-single \
  --i-demultiplexed-seqs demux.qza \
  --p-trim-left 0 \
  --p-trunc-len 120 \
  --o-representative-sequences rep-seqs-dada2.qza \
  --o-table table-dada2.qza \
  --o-denoising-stats stats-dada2.qza
# Output artifacts
qiime metadata tabulate \
  --m-input-file stats-dada2.qza \
  --o-visualization stats-dada2.qzv
# Visualize at view.qiime2.org
# Rename the files
mv rep-seqs-dada2.qza rep-seqs.qza

#Generating a Phylogenetic tree
qiime phylogeny align-to-tree-mafft-fasttree \
  --i-sequences rep-seqs.qza \
  --o-alignment aligned-rep-seqs.qza \
  --o-masked-alignment masked-aligned-rep-seqs.qza \
  --o-tree unrooted-tree.qza \
  --o-rooted-tree rooted-tree.qza

# Generate Alpha and Beta Diversity Numbers using a Sampling Depth of 1103
# Scroll down to "Alpha and beta diveristy analysis"
# We can change the sampling depth ("1103" - the minimum length of the reads it will pull) If you pick too low of a number, you might miss a lot of diveristy in the larger reads, but if you choose a numbe too high, it will exclude the reads that are not long enough
qiime diversity core-metrics-phylogenetic \
  --i-phylogeny rooted-tree.qza \
  --i-table table.qza \
  --p-sampling-depth 1103 \
  --m-metadata-file sample-metadata.tsv \
  --output-dir core-metrics-results

# Differential Abundance Testing with ANCOM-BC
# Scroll down to 'Differential Abundance Testing with ANCOM-BC' in QIIME tutorial
# Filter for gut-only
qiime feature-table filter-samples \
  --i-table table.qza \
  --m-metadata-file sample-metadata.tsv \
  --p-where "[body-site]='gut'" \
  --o-filtered-table gut-table.qza
# Differential Abundance (between the two subjects)
qiime composition ancombc \
  --i-table gut-table.qza \
  --m-metadata-file sample-metadata.tsv \
  --p-formula 'subject' \
  --o-differentials ancombc-subject.qza

qiime composition da-barplot \
  --i-data ancombc-subject.qza \
  --p-significance-threshold 0.001 \
  --o-visualization da-barplot-subject.qzv
# Collapse to a specific taxonomic level (Abundance at genus level)
qiime taxa collapse \
  --i-table gut-table.qza \
  --i-taxonomy taxonomy.qza \
  --p-level 6 \
  --o-collapsed-table gut-table-l6.qza

qiime composition ancombc \
  --i-table gut-table-l6.qza \
  --m-metadata-file sample-metadata.tsv \
  --p-formula 'subject' \
  --o-differentials l6-ancombc-subject.qza

qiime composition da-barplot \
  --i-data l6-ancombc-subject.qza \
  --p-significance-threshold 0.001 \
  --p-level-delimiter ';' \
  --o-visualization l6-da-barplot-subject.qzv
