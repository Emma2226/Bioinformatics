# Prokka
# Activate genome assembly environment
mamba activate genomeassembly
# Install Prokka (if not done already)
mamba install -c conda-forge -c bioconda -c defaults prokka
# Navigate into the genome assembly folder you are working in
cd /mnt/c/Users/eh040/Desktop/Bioinformatics/GenomeAssemblyandAnnotationReport
# Run Prokka ('prokka'=calls the program; '--outdir ProkkaAnnotation'=creates an output directory that is called 'ProkkaAnnotation'; '--prefix klebsiella'=gives the files a prefix name 'kebsiella'; 'ABySSOutput/unknowngenome-8.fa'=tells the program which file we want it to annotate in the ABySSOutput folder)
prokka --outdir ProkkaAnnotation --prefix klebsiella ABySSOutput/unknowngenome-8.fa
